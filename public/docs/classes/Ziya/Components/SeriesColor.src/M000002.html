    def flatten( xml )
      unless colors.nil?
        xml.series_color do
          if colors.is_a? String
            cols = colors.split( "," )
            cols.each { |c| xml.color( c.strip ) }
          elsif colors.respond_to? :each
            colors.each { |c| xml.color( c ) }
          else
            xml.color( colors )
          end
        end   
      end
    end
