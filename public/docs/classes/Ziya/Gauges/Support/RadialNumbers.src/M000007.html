    def flatten( xml )
      number = start_num
      i      = start_angle
      while( i <= end_angle ) do
        angle = deg2rad( i )
        hash = { :x         => (x + Math::sin( angle ) * radius).to_i,
                 :y         => (y - Math::cos( angle ) * radius).to_i,
                 :width     => width ||  20,
                 :height    => height || 20,
                 :font      => font || "Arial",
                 :size      => size || 10,
                 :bold      => bold || true,
                 :alpha     => alpha,                 
                 :rotation  => i, 
                 :align     => align || "left",
                 :color     => color }
        xml.text( hash ) do
          xml.text! number.to_s
        end
        i      += (end_angle-start_angle)/(ticks-1)
        number += (end_num-start_num)/(ticks-1)
      end
    end
