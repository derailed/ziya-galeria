      def render_extra_components
        if options[:url]
          Ziya::Gauges::Support::Update.new(
            :url        => options[:url],
            :retry      => options[:retry] || 0,
            :timeout    => options[:timeout] || 30,
            :delay_type => options[:delay_type] || 1,
            :delay      => options[:delay] || 30 ).flatten( @xml )
        end
      end
